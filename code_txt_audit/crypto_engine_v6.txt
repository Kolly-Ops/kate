def ts_print(msg):
    print(f"[{datetime.now().strftime('%H:%M:%S')}] {msg}")

def check_singleton():
    lock_path = "kate_crypto.lock"
    if os.path.exists(lock_path):
        try: os.remove(lock_path)
        except:
            ts_print("[CRITICAL] KATE is already running. Duplicate process blocked.")
            sys.exit(0)
    open(lock_path, 'w').close()

def cleanup_lock():
    lock_path = "kate_crypto.lock"
    if os.path.exists(lock_path):
        try: os.remove(lock_path)
        except: pass

def calculate_rsi(prices, period=14):
    """Calculate RSI from a list of prices."""
    if len(prices) < period + 1:
        return 50  # Not enough data, return neutral
    
    deltas = [prices[i] - prices[i-1] for i in range(1, len(prices))]
    gains = [d if d > 0 else 0 for d in deltas[-period:]]
    losses = [-d if d < 0 else 0 for d in deltas[-period:]]
    
    avg_gain = sum(gains) / period
    avg_loss = sum(losses) / period
    
    if avg_loss == 0:
        return 100
    rs = avg_gain / avg_loss
    return 100 - (100 / (1 + rs))

class Position:
    def __init__(self):
        self.active = False
        self.symbol = ""
        self.bullets_fired = 0
        self.total_qty = 0.0
        self.avg_price = 0.0
        self.entry_time = 0
        self.last_dca_price = 0.0
        self.trade_capital = 0.0
        self.trailing_peak = 0.0
        self.is_trailing = False

    def to_dict(self):
        return self.__dict__

    def from_dict(self, d):
        for k, v in d.items():
            setattr(self, k, v)

    def save(self):
        with open("position_state.json", "w") as f:
            json.dump(self.to_dict(), f)

    def load(self):
        if os.path.exists("position_state.json"):
            try:
                with open("position_state.json", "r") as f:
                    d = json.load(f)
                    self.from_dict(d)
                return self.active
            except: pass
        return False

    def add_execution(self, price, qty):
        if self.total_qty + qty > 0:
            current_value = self.total_qty * self.avg_price
            new_value = qty * price
            self.avg_price = (current_value + new_value) / (self.total_qty + qty)
        else:
            self.avg_price = price
            
        self.total_qty += qty
        self.last_dca_price = price
        self.bullets_fired += 1
        
        if not self.active:
            self.active = True
            self.entry_time = time.time()
            self.trailing_peak = price
        self.save()

    def close(self):
        self.active = False
        self.symbol = ""
        self.bullets_fired = 0
        self.total_qty = 0.0
        self.avg_price = 0.0
        self.entry_time = 0
        self.trade_capital = 0.0
        self.trailing_peak = 0.0
        self.is_trailing = False
        if os.path.exists("position_state.json"):
            os.remove("position_state.json")

current_pos = Position()

# --- WEBSOCKET LISTENERS ---
def on_liq_message(ws, message):
    data = json.loads(message)
    try:
        payload = data.get('o', {})
        symbol = payload.get('s', '').replace('USDT', '')
        if symbol in WATCH_SYMBOLS and payload.get('S') == "SELL":
            q, p = float(payload.get('q', 0)), float(payload.get('p', 0))
            with lock:
                symbols_state[symbol]['liqs_deque'].append((time.time(), q * p))
    except: pass

def on_ticker_message(ws, message):
    data = json.loads(message)
    try:
        # Handles both single and multiple stream formats
        results = data if isinstance(data, list) else [data]
        for item in results:
            s_data = item.get('data', item)
            raw_sym = s_data.get('s', '')
            symbol = raw_sym.replace('USDT', '')
            symbol = symbol.replace('USDT', '')
            if symbol in WATCH_SYMBOLS:
                price = float(s_data.get('c', 0))
                with lock:
                    symbols_state[symbol]['price'] = price
                    symbols_state[symbol]['price_history'].append(price)
                    engine_meta['last_update_ts'] = time.time()
    except: pass

def start_websocket_streams():
    # 1. Liquidation Stream (Futures)
    liq_url = "wss://fstream.binance.com/ws/!forceOrder@arr"
    def run_liq():
        while True:
            try:
                ws = websocket.WebSocketApp(liq_url, on_message=on_liq_message)
                ws.run_forever()
            except: pass
            time.sleep(5) # Auto-retry
    threading.Thread(target=run_liq, daemon=True).start()

    # 2. Ticker Stream (Spot)
    streams = "/".join([f"{s.lower()}usdt@ticker" for s in WATCH_SYMBOLS])
    ticker_url = f"wss://stream.binance.com:9443/stream?streams={streams}"
    def run_ticker():
        while True:
            try:
                ws = websocket.WebSocketApp(ticker_url, on_message=on_ticker_message)
                ws.run_forever()
            except: pass
            time.sleep(5) # Auto-retry
    threading.Thread(target=run_ticker, daemon=True).start()

def get_symbol_liqs(symbol, seconds=60):
    now = time.time()
    with lock:
        dq = symbols_state[symbol]['liqs_deque']
        while dq and dq[0][0] < now - seconds:
            dq.popleft()
        return sum(v for _, v in dq)

def log_trade(side, price, qty, pnl=0, note=""):
    log_file = "trade_history.csv"
    exists = os.path.exists(log_file)
    try:
        with open(log_file, "a", newline="") as f:
            writer = csv.writer(f)
            if not exists:
                writer.writerow(["timestamp", "side", "price", "qty", "pnl_usd", "note"])
            # Estimate Fees (0.1% per trade)
            est_fee_usd = (price * qty) * 0.001
            # For BUY: Record 0 profit, but we effectively 'paid' the fee from our cash
            # For SELL: Record Gross PnL - Sell Fee - estimated Buy Fee
            net_pnl = pnl - est_fee_usd if side == "SELL" else -est_fee_usd
            writer.writerow([datetime.now().strftime("%Y-%m-%d %H:%M:%S"), side, price, qty, f"{net_pnl:.4f}", note])
    except Exception as e:
        ts_print(f"Logging Error: {e}")

def check_and_send_alert(pnl_usd, symbol):
    if abs(pnl_usd) >= 5.0:
        subject = f"🚨 KATE MAJOR ACTIVITY: {symbol} ${pnl_usd:+.2f}"
        body = f"""
        <h3>Major Trade Activity Detected</h3>
        <p><b>Symbol:</b> {symbol}</p>
        <p><b>PnL:</b> <span style="color: {'green' if pnl_usd >= 0 else 'red'}">${pnl_usd:.2f}</span></p>
        <p>Timestamp: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}</p>
        """
        send_alert(subject, body)

# --- EXECUTION HANDLER (100% SIMULATION LOCK) ---
def execute_trade(exchange, side, usd_amount, spot_price, is_paper, symbol_pair, exit_reason=""):
    global current_pos
    qty = usd_amount / spot_price
    
    # HARD SAFETY LOCK: Ensure no real orders can EVER be sent
    if not is_paper:
        ts_print(f"[SAFETY BLOCK] BLOCKED {side} order for {symbol_pair}. LIVE TRADING IS GLOBALLY DISABLED.")
        return 0

    if side == "BUY":
        ts_print(f"[SIMULATION] BUYING ${usd_amount:.2f} {symbol_pair} @ {spot_price}")
        current_pos.add_execution(spot_price, qty)
        log_trade("BUY", spot_price, qty, 0, f"[SIM] Bullet fired on {symbol_pair}")
    elif side == "SELL":
        pnl_usd = (spot_price - current_pos.avg_price) * current_pos.total_qty
        ts_print(f"[SIMULATION] SELLING ALL {symbol_pair} @ {spot_price}. PnL: ${pnl_usd:.2f}")
        log_trade("SELL", spot_price, current_pos.total_qty, pnl_usd, f"[SIM] Exit on {symbol_pair} ({exit_reason})")
        current_pos.close()
        return pnl_usd
    return 0

                with lock: spot_price = symbols_state[sym]['price']
                if spot_price == 0: continue

                        update_pnl('crypto_pnl', cumulative_pnl)
                        check_and_send_alert(pnl_usd, sym)
                        current_pos.close()
                        is_in_position = False
                        continue

                # Apply 12h stagnation rule to everything EXCEPT the current DOGE session
                is_stagnant = (elapsed > STAGNATION_HOURS * 3600) and (trade_pnl_usd < 0) and (sym != "DOGE")
                
                # FINAL DEADLINE: Cut any open trade at 19:45 (7:45 PM) Local Time to ensure EOD flatness
                now_time = datetime.now().time()
                is_deadline = (now_time.hour == 19 and now_time.minute >= 45) or (now_time.hour >= 20)
                
                if pnl_pct < -STOP_LOSS_PCT or trade_pnl_usd <= -MAX_DRAWDOWN_USD or is_stagnant or is_deadline:
                    if is_deadline:
                        note = "EOD EXIT (Flat for tomorrow)"
                    elif is_stagnant:
                        note = "STAGNATION EXIT (12h+)"

                        note = "MAX DRAWDOWN ($%s)" % MAX_DRAWDOWN_USD
                    else:
                        note = "STOP LOSS %"

                    pnl_usd = execute_trade(spot_exchange, "SELL", 0, spot_price, active_paper_mode, f"{sym}/USDT", exit_reason=note)
                    cumulative_pnl += pnl_usd
                    update_pnl('crypto_pnl', cumulative_pnl)
                    check_and_send_alert(pnl_usd, sym)
                    current_pos.close()
                    is_in_position = False

                    if drop_from_last < -DCA_DROP_TRIGGER:
                        ts_print(f"!!! {sym} DCA Triggered (-{drop_from_last*100:.2f}%) !!!")
                        amount = current_pos.trade_capital * DCA_BULLETS[current_pos.bullets_fired]
                        execute_trade(spot_exchange, "BUY", amount, spot_price, active_paper_mode, f"{sym}/USDT")
                
                # D. CIRCUIT BREAKER
                if elapsed > CIRCUIT_BREAKER_TIME and pnl_pct < CIRCUIT_BREAKER_TARGET:
                        ts_print(f"!!! {sym} CIRCUIT BREAKER EXIT !!!")
                        pnl_usd = execute_trade(spot_exchange, "SELL", 0, spot_price, active_paper_mode, f"{sym}/USDT", exit_reason="Circuit Breaker")

                    if liqs > current_threshold and oi_chg < -OI_DROP_PCT:

                        rsi = calculate_rsi(price_list)
                        if rsi >= 28:
                            if time.time() % 60 < 1:  # Log once per minute
                                ts_print(f"[FILTER] {sym} signal rejected: RSI={rsi:.1f} (need <32)")
                            continue

                        # --- KILL SWITCH CHECK (Graceful) ---
                        if get_kill_switch_status():
                            if time.time() % 300 < 1: # Log every 5 mins
                                ts_print("[SAFE MODE] KILL SWITCH ACTIVE. Skipping new entry signal.")
                            continue

                        # SET FLAG IMMEDIATELY to prevent double firing in current loop
                        is_in_position = True
                        
                        ts_print(f">>> SIGNAL: {sym} | Liq ${liqs:,.0f} | OI Drop {oi_chg*100:.2f}% <<<")
                        
                        # FLUSH LIQS for this symbol to avoid "Machine Gun" firing on next tick
                        with lock:
                            symbols_state[sym]['liqs_deque'].clear()

                        # Dynamic Compounding Capital Fetch
                        cap = 0
                        if spot_exchange:
                            try:
                                # Fetch fresh balance to compound profits
                                bal = spot_exchange.fetch_balance()
                                cap = bal['free'].get('USDT', 0.0)
                                if cap < 10.0: # Filter for minimum trade size
                                    ts_print(f"[WARN] Balance too low to trade: ${cap:.2f}")
                                    continue
                            except Exception as e:
                                ts_print(f"[ERROR] Could not fetch balance for compounding: {e}")
                                continue
                        else:
                            cap = TOTAL_CAPITAL_USD # Fallback for PAPER mode
                        
                        current_pos.symbol = sym
                        current_pos.trade_capital = cap
                        amount = cap * DCA_BULLETS[0]
                        ts_print(f">>> COMPOUNDING: Firing 20% of Current Balance (${amount:.2f}) <<<")
                        execute_trade(spot_exchange, "BUY", amount, price, active_paper_mode, f"{sym}/USDT")

                if not is_in_position and time.time() % 60 < 1:
                    status_msg = "[PULSE] KATE is alive and monitoring..."
                    if get_kill_switch_status():
                        status_msg = "[SAFE MODE] KILL SWITCH ACTIVE. Halting new entries."
                    ts_print(status_msg)

        except Exception as e:
            ts_print(f"Engine Error: {e}")
            time.sleep(1)
        
        time.sleep(0.5) # Fast loop for WebSocket processing
